N - Number of keys in a Node
M - Number of key/value pairs in a Leaf.

NodeSize = N * sizeof(Key) + sizeof(void*)
NodeGroupSize = (N + 1) * NodeSize
LeafSize = M * (sizeof(Key) + sizeof(Value))
LeafGroupSize = (N + 1) * LeafSize + 2 * sizeof(void*)

I - Number of key/value pairs to store

NumberOfLeafs = up(I / M)
NumberOfNodes = NumberOfLeafs / (N + 1) + log_N+1(NumberOfLeafs / (N + 1))
